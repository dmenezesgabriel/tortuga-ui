import variables from "@/stories/design/design.module.scss";
import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs";

<Meta title="Design System/Colors" />

import { colorGroups, colorVariants } from "./design";

# Tokens

<ColorPalette>
  {colorGroups.map((colorName) => {
    let colorVariantsObject = {};
    colorVariants.forEach((variantName) => {
      colorVariantsObject[`${colorName}-${variantName}`] =
        variables[`${colorName}-${variantName}`];
    });
    return (
      <ColorItem
        key={colorName}
        title={colorName}
        colors={colorVariantsObject}
      />
    );
  })}
</ColorPalette>
